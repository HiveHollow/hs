  /* testunion */ |* \* —* /* |* \* —*



  struct v2 {
    float x, y;
  };

  struct v4 {
    union { // anonymous union
  
      struct { // anonymous struct
    
        float x, y, z, w;
      
      };
      struct { // anonymous struct
    
        v2 a, b;
      
      };
    };
  };

  void vec_ptwo(const v2 &vector) {
    cout << vector.x << " , " << vector.y << endl;
  }

  int main() {
    v4 vector = {1.0f, 2.0f, 3.0f, 4.0f};
    vec_ptwo(vector.a);
    vec_ptwo(vector.b);
    vector.z = 11.0f;
    cout << endl;
    vec_ptwo(vector.a);
    vec_ptwo(vector.b);
    cout << vector.z << endl;
    cout << vector.b.x << endl;
  
    return 0;
  }

  // z corresponds to b.x   occupies the same place in memory
