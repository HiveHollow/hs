       /* Example 1 */ [In this example we use referenced variables]



   int ingredient;
   void function(int &var);


   int main()
       {
           ingredient = 3;

           function(ingredient);
           return 0;
       }
    
   void function(int &var)
       {
           cout << "equals " << var << "\n";

           ingredient = ingredient + 4;
           cout << "equals " << var << "\n";
           ingredient = 5;
           cout << "equals " << var << "\n";
           ingredient = ingredient - 1;
           cout << "equals " << var << "\n";
       }



       /* Example 2 */ [lvalue and rvalue]



   #define local static // words like 'object' and 'static' can be misinterpreted,
                        so replace them in a macro

   int &v() { // an lvalue reference
     local int v = 3;
     return v;
   }

   int main() {
     int i = v(); // v 'return's a temporary value(no location or storage), an rvalue
     // int i is an lvalue
     v() = 4; // an lvalue reference, is now returned
  
     cout << v();
     cout << i;
  
     return 0;
}
